import com.user.test.entitys.Security;
import com.user.test.service.SecurityService;
import org.springframework.web.bind.annotation.*;
import org.springframework.beans.factory.annotation.Autowired;
import java.util.List;
import java.util.Date;
@RestController
@CrossOrigin(origins = "*")
@RequestMapping("/security")
public class SecurityController {
;
SecurityService securityService;
@GetMapping("/Getcode/{code}")
private Security findByCode(@PathVariable("code") Long code) {
return securityService.findByCode(code);
}
@GetMapping("/Getcodecontain/{code}")
private List<Security> findByCodeContain( @PathVariable("code") Long code){
return securityService.findByCodeContaining(code);
}
@GetMapping("/GetSecurity/{id}")
private List<Security> findByIdSecurity( @PathVariable("id") Long id){
return securityService.findById(id);
}
@GetMapping("/GetAllSecurity")
private List<Security> getAllSecurity(){
return securityService.getAllSecurity();
}
@PostMapping("/saveOrUpdate")
private Boolean saveOrUpdateSecurity( @RequestBody Security security){
return securityService.saveOrUpdateSecurity(security);
}
@GetMapping("/Search")
private List<Security> finBySearchSecurity( @RequestParam(value = "search") String search){
return securityService.search(search);
}
